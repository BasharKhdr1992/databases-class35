1.1.1. values in column dinner_date are stored in different formats.
1.1.2. food_code, and food_description columns store multiple values.

1.2. Entities are ---> members, dinners, venues, food.

1.3. The following schema is a normalized version of the given dinner club database:

create database dinners;

use dinners;

create table members (
	member_id int primary key,
    member_name varchar(50) not null,
    member_address varchar(50)
);

create table venues (
	venue_code varchar(5) primary key,
    venue_description varchar(100) not null
);

create table dinners (
	dinner_id varchar(15) primary key,
    dinner_date datetime,
    venue_code varchar(5) not null,
    foreign key (venue_code) references venues(venue_code)
);

create table members_dinners_bridge (
	member_id int,
    dinner_id varchar(15),
    primary key (member_id, dinner_id),
    foreign key (member_id) references members(member_id),
    foreign key (dinner_id) references dinners(dinner_id)
);

create table food (
	food_code varchar(5) primary key,
    food_description varchar(30) not null
);

create table food_dinner_bridge (
	food_code varchar(5),
    dinner_id varchar(15),
    primary key (food_code, dinner_id),
    foreign key (food_code) references food(food_code),
    foreign key (dinner_id) references dinners(dinner_id)
);
